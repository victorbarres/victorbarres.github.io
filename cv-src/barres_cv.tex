\section*{Experience}
% ============================================================================

\position{Sierra}{San Francisco, CA}{Sep 2024 – Present}
  {AI Researcher}{}
\begin{itemize}
  \item Lead the \textbf{$\tau$-Bench family} of agent benchmarks --- research direction, codebase, public leaderboard (\href{https://taubench.com}{taubench.com}), and ecosystem of extensions: $\tau^2$-Bench (dual control), $\tau$-Knowledge (knowledge retrieval), $\tau$-Voice (full-duplex voice), and $\tau^3$-Bench (the combined release with community-contributed task and code improvements). Three papers from the family accepted to \textbf{ICML 2026}.
  \item Kickstarted Sierra's internal \textbf{agent evaluation system}: set up the proof of concept and did the original work on user-simulation-driven stress testing for pre-release agent hardening across deployed customer-service domains.
  \item Organizer \& judge of the Sierra $\tau^2$-Bench Custom Track at the \href{https://rdi.berkeley.edu/agentx-agentbeats.html}{AgentX–AgentBeats} competition (Berkeley RDI, Fall 2025 – Spring 2026).
\end{itemize}

\position{Elemental Cognition}{Remote}{Jun 2023 – Jul 2024}
  {Senior Researcher (NLP)}{}
\begin{itemize}
  \item Developed core algorithms for EC's hybrid AI architecture, integrating LLMs with formal reasoning engines.
  \item Led the integration of multi-round RAG with formal models of causation for knowledge-intensive multi-hop QA, published as \emph{From Generating Answers to Building Explanations} at NAACL 2025 (Industry).
  \item Built conversational agents for two reasoning-heavy settings: a logic-programming assistant that elicits and translates user requirements into a logic program, and an ASP-reasoner-backed constraint-solving agent.
\end{itemize}

\position{Uniphore.ai}{Palo Alto, CA}{Dec 2019 – Jun 2023}
  {Principal NLP Scientist \& NLP Research Lead}{previously Senior NLP Scientist (2019–2021)}
\begin{itemize}
  \item Built and led Uniphore's NLP Lab (7+ scientists). Set short-, mid-, and long-term research strategy for understanding human–human spoken conversation in the contact-center domain.
  \item Shipped foundational and applied work across LLM benchmarking, prompt engineering, structured knowledge grounding, summarization, semantic parsing, dialogue-act and dialogue-flow modeling, and real-time caller-intent recognition.
  \item Kickstarted the team, deployed first NER / redaction / call-driver models, and stood up cloud infrastructure and CI/CD for production NLP.
\end{itemize}

\position{Astound.ai}{Menlo Park, CA}{Sep 2017 – Dec 2019}
  {AI Scientist (NLP)}{}
\begin{itemize}
  \item Core member of the team that built, from the ground up, Astound's domain-expert conversational agent for IT / HR / Finance enterprise self-service.
  \item Led the neuro-symbolic side of the agent's NLP stack — word/sentence embeddings, semantic parsing, knowledge-graph embeddings, and semantic inference — integrated with a modular dialogue manager (domain classification, intent recognition, search, small talk, sub-flows).
  \item Built NLP-side APIs over Astound's ontology and extended it to support inference-driven multi-turn clarification, establishing a separation of concerns between knowledge, lexicon, and grammar engineers.
\end{itemize}
